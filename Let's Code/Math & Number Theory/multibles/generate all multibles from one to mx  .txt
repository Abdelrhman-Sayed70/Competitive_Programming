#include<bits/stdc++.h>
using namespace std;
#define ll long long 

void Generating_all_multiples(int mx) {
    for (int i = 1; i <= mx; i++) {
        for (int j = i; j <= mx; j += i) {
            cout << j << ' ';
        }
        cout << '\n';
    }
}
int main() {
    // this function generate all multibles of each number from 1 to mxnumber in nearly O(n) >> used as preprocessing
    Generating_all_multiples(5);    
}
